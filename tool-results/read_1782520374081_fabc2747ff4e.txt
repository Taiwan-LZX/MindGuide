   154→function buildMessageContext(
   155→  userMessages: Array<{ role: string; content: string; type?: string }>,
   156→  knowledgeNodes: Array<{ title: string; content: string; category?: string; mastered: boolean }>,
   157→  kbContext: string,
   158→  teachingMode: string = 'guide',
   159→  thinkingMode: string = 'standard'
   160→) {
   161→  const contextParts: Array<{ role: 'system' | 'user' | 'assistant'; content: string }> = [];
   162→
   163→  const thinkOverlay = THINK_OVERLAYS[thinkingMode] || '';
   164→  contextParts.push({
   165→    role: 'system',
   166→    content: TEACHING_SYSTEM_PROMPT + (MODE_OVERLAYS[teachingMode] || MODE_OVERLAYS.guide) + thinkOverlay,
   167→  });
   168→
   169→  if (knowledgeNodes.length > 0) {
   170→    const masteredNodes = knowledgeNodes.filter((n) => n.mastered);
   171→    const unmasteredNodes = knowledgeNodes.filter((n) => !n.mastered);
   172→
   173→    let knowledgeContext = '## 学习者的知识状态\n\n';
   174→
   175→    if (masteredNodes.length > 0) {
   176→      knowledgeContext += '已掌握的知识点：\n';
   177→      masteredNodes.forEach((n) => {
   178→        knowledgeContext += `- ${n.title}: ${n.content.slice(0, 100)}...\n`;
   179→      });
   180→    }
   181→
   182→    if (unmasteredNodes.length > 0) {
   183→      knowledgeContext += '\n正在学习/未掌握的知识点：\n';
   184→      unmasteredNodes.forEach((n) => {
   185→        knowledgeContext += `- ${n.title} (${n.category || '概念'}): ${n.content.slice(0, 100)}...\n`;
   186→      });
   187→    }
   188→
   189→    contextParts.push({
   190→      role: 'system',
   191→      content: knowledgeContext,
   192→    });
   193→  }
   194→
   195→  // ─── Knowledge base: RAG-retrieved passages ──────────────────────────────
   196→  //
   197→  // Instead of brute-force injecting the first 20k chars of all materials,
   198→  // we retrieve the top-K passages most relevant to the learner's latest
   199→  // message. This gives the model focused, cited context (Lost in the Middle
   200→  // mitigation — Liu et al. 2023).
   201→  if (kbContext) {
   202→    contextParts.push({
   203→      role: 'system',
   204→      content: kbContext,
   205→    });
   206→  }
   207→
   208→  // ─── Token-budget-aware history compression ──────────────────────────────
   209→  //
   210→  // Older versions just took the last 20 messages verbatim. That overflows
   211→  // the context window on long conversations (each turn can be 1-4k tokens),
   212→  // causing the API to truncate silently or reject the request. We now:
   213→  //
   214→  //   1. Estimate tokens per message (~chars/4 for CJK + ASCII mix).
   215→  //   2. Walk from the MOST RECENT message backwards, accumulating until we
   216→  //      hit a soft budget (RESERVED_TOKENS for system+KB+reply headroom).
   217→  //   3. If even the recent messages overflow, we keep the most recent N
   218→  //      fully and FOLD older messages into a one-line summary each
   219→  //      ("用户问了 X，助教回答了 Y 的要点") so the thread stays coherent
   220→  //      without blowing the budget.
   221→  //   4. The very first user turn is ALWAYS kept in full (it sets the topic).
   222→  //
   223→  // This is a pragmatic middle ground between "drop everything old" (loses
   224→  // topic context) and "keep everything" (overflows). RAG already handles
   225→  // long-term knowledge retrieval; this compression only governs the
   226→  // conversational thread.
   227→
   228→  const MODEL_CONTEXT_WINDOW = 200_000;
   229→  const RESERVED_FOR_SYSTEM_KB_REPLY = 24_000; // headroom for system + KB + the new reply
   230→  const historyBudget = MODEL_CONTEXT_WINDOW - RESERVED_FOR_SYSTEM_KB_REPLY;
   231→
   232→  // Rough token estimator: CJK chars ≈ 1 token each, ASCII ≈ 0.25 tokens/char.
   233→  const estTokens = (s: string) => {
   234→    let cjk = 0, other = 0;
   235→    for (const ch of s) {
   236→      const code = ch.codePointAt(0) || 0;
   237→      if ((code >= 0x3000 && code <= 0x9fff) || (code >= 0xff00 && code <= 0xffef)) cjk++;
   238→      else other++;
   239→    }
   240→    return cjk + Math.ceil(other / 4);
   241→  };
   242→
   243→  // Pre-compute per-message token estimates.
   244→  const tagged = userMessages.map((m, i) => ({ msg: m, idx: i, tokens: estTokens(m.content) }));
   245→
   246→  // Always keep the first user turn (topic anchor) — it's almost always short.
   247→  const firstUserTurn = tagged.find(t => t.msg.role === 'user');
   248→  const firstKeepIds = new Set<number>(firstUserTurn ? [firstUserTurn.idx] : []);
   249→
   250→  // Walk backwards from the latest, accumulating full messages until budget.
   251→  const keptFull: typeof tagged = [];
   252→  let usedTokens = firstUserTurn ? firstUserTurn.tokens : 0;
   253→  for (let i = tagged.length - 1; i >= 0; i--) {
   254→    const t = tagged[i];
   255→    if (firstKeepIds.has(i)) continue; // already counted
   256→    if (usedTokens + t.tokens > historyBudget) break;
   257→    keptFull.unshift(t);
   258→    usedTokens += t.tokens;
   259→  }
   260→  // Ensure first turn is in the kept set (at the front).
   261→  if (firstUserTurn) keptFull.unshift(firstUserTurn);
   262→
   263→  // Anything older than keptFull that isn't the first turn gets folded.
   264→  const keptIdx = new Set(keptFull.map(t => t.idx));
   265→  const folded = tagged.filter(t => !keptIdx.has(t.idx));
   266→
   267→  // Build a single system summary of folded turns (if any).
   268→  if (folded.length > 0) {
   269→    const summaryLines = folded.map(t => {
   270→      const role = t.msg.role === 'assistant' ? '助教' : '学习者';
   271→      const snippet = t.msg.content.replace(/\s+/g, ' ').slice(0, 60);
   272→      return `- ${role}：${snippet}`;
   273→    });
   274→    contextParts.push({
   275→      role: 'system',
   276→      content: `## 更早的对话摘要（已折叠以节省上下文）\n${summaryLines.join('\n')}`,
   277→    });
   278→  }
   279→
   280→  // Normalize roles, then MERGE consecutive same-role messages into one.
   281→  // The ZAI Chat Completions API rejects message arrays where two adjacent
   282→  // turns share the same role (error code 1214 "messages 参数非法").
   283→  // This happens naturally when the learner sends two messages in a row, or
   284→  // when an assistant turn was saved in two pieces — so we coalesce them
   285→  // with a blank-line separator before sending.
   286→  for (const t of keptFull) {
   287→    const role = (t.msg.role === 'assistant' ? 'assistant' : t.msg.role === 'system' ? 'system' : 'user') as
   288→      | 'system'
   289→      | 'user'
   290→      | 'assistant';
   291→    const last = contextParts[contextParts.length - 1];
   292→    if (last && last.role === role) {
   293→      last.content = `${last.content}\n\n${t.msg.content}`;
   294→    } else {
   295→      contextParts.push({ role, content: t.msg.content });
   296→    }
   297→  }
   298→
   299→  // Guarantee the conversation ends with a user turn — the API requires the
   300→  // final message to be from the user when requesting a completion.
   301→  while (contextParts.length > 1 && contextParts[contextParts.length - 1].role !== 'user') {
   302→    // Drop trailing assistant/system turns that have no user follow-up
   303→    // (rare, but avoids "messages must end with user" errors).
   304→    if (contextParts[contextParts.length - 1].role === 'system') break;
   305→    contextParts.pop();
   306→  }
   307→
   308→  return contextParts;
   309→}

   311→// Parse an upstream SSE byte chunk and extract delta content pieces.
   312→// Buffers incomplete lines and returns them via the carried-over buffer string.
   313→//
   314→// Returns an array of { kind: 'thinking' | 'content', text } tokens — the GLM
   315→// chat-completions stream emits `delta.reasoning_content` during the model's
   316→// internal reasoning phase (only when thinking is enabled) and
   317→// `delta.content` during the answer phase. Surfacing both lets the client
   318→// render a proper "思考中" animation that persists for the WHOLE reasoning
   319→// phase, not just the pre-first-token gap.
   320→function parseUpstreamSse(
   321→  chunkStr: string,
   322→  bufferRef: { buf: string }
   323→): Array<{ kind: 'thinking' | 'content'; text: string }> {
   324→  bufferRef.buf += chunkStr;
   325→  const pieces: Array<{ kind: 'thinking' | 'content'; text: string }> = [];
   326→  // SSE events are separated by a blank line (\n\n)
   327→  const events = bufferRef.buf.split('\n\n');
   328→  // The last element is the incomplete tail (no trailing \n\n yet)
   329→  bufferRef.buf = events.pop() || '';
   330→
   331→  for (const evt of events) {
   332→    const lines = evt.split('\n');
   333→    for (const line of lines) {
   334→      const trimmed = line.trim();
   335→      if (!trimmed.startsWith('data:')) continue;
   336→      const data = trimmed.slice(5).trim();
   337→      if (!data || data === '[DONE]') continue;
   338→      try {
   339→        const parsed = JSON.parse(data);
   340→        const delta = parsed?.choices?.[0]?.delta;
   341→        if (!delta || typeof delta !== 'object') continue;
   342→        // Reasoning content (the model's internal chain-of-thought). Present
   343→        // only when thinking is enabled. We forward it as a separate token
   344→        // kind so the client can drive a thinking-phase animation.
   345→        const reasoning = delta.reasoning_content;
   346→        if (typeof reasoning === 'string' && reasoning.length > 0) {
   347→          pieces.push({ kind: 'thinking', text: reasoning });
   348→        }
   349→        // Answer content (the visible reply).
   350→        const content = delta.content;
   351→        if (typeof content === 'string' && content.length > 0) {
   352→          pieces.push({ kind: 'content', text: content });
   353→        }
   354→      } catch {
   355→        // skip malformed JSON
   356→      }
   357→    }
   358→  }
   359→  return pieces;
   360→}

   362→export async function POST(req: NextRequest) {
   363→  try {
   364→    const parsed = await parseBody(req, chatSchema);
   365→    if (!parsed.ok) return parsed.response;
   366→    const { sessionId, message, messages: historyMessages, knowledgeNodes, teachingMode, thinkingMode, selectedModel } = parsed.data;
   367→
   368→    // Save user message to database
   369→    await db.learningMessage.create({
   370→      data: {
   371→        sessionId,
   372→        role: 'user',
   373→        content: message,
   374→        type: 'dialogue',
   375→      },
   376→    });
   377→
   378→    // Update session timestamp
   379→    await db.learningSession.update({
   380→      where: { id: sessionId },
   381→      data: { updatedAt: new Date() },
   382→    });
   383→
   384→    // Prepare messages for AI
   385→    const zai = await ZAI.create();
   386→
   387→    // ─── RAG retrieval: find passages most relevant to the learner's message ─
   388→    //
   389→    // We retrieve from the session's knowledge base (all uploaded materials
   390→    // that have been parsed + chunked + embedded). This replaces the old
   391→    // brute-force "inject first 20k chars of every material" approach.
   392→    //
   393→    // The learner's latest message is the natural query — it's what they're
   394→    // asking about right now. We pass it through retrievePassages() which
   395→    // embeds it, scans all chunk embeddings, and returns the top-K with
   396→    // parent-section text attached.
   397→    const passages = await retrievePassages(sessionId, message, 6);
   398→    const kbContext = buildKnowledgeBaseContext(passages, 18_000);
   399→
   400→    const messageHistory = buildMessageContext(historyMessages || [], knowledgeNodes || [], kbContext, teachingMode, thinkingMode);
   401→
   402→    // Real streaming: SDK returns the upstream ReadableStream when stream:true
   403→    const upstream = (await zai.chat.completions.create({
   404→      messages: messageHistory,
   405→      model: selectedModel,
   406→      thinking: thinkingConfig(thinkingMode),
   407→      stream: true,
   408→    } as any)) as ReadableStream<Uint8Array> | undefined;
   409→
   410→    // Fallback: if the SDK didn't return a stream (unexpected), use non-streaming
   411→    if (!upstream || typeof upstream.getReader !== 'function') {
   412→      const completion = await zai.chat.completions.create({
   413→        messages: messageHistory,
   414→        model: selectedModel,
   415→        thinking: thinkingConfig(thinkingMode),
   416→      });
   417→      const aiContent: string = stripEmoji(completion?.choices?.[0]?.message?.content || '');
   418→      if (!aiContent) {
   419→        return new Response(JSON.stringify({ error: 'Empty AI response' }), {
   420→          status: 500,
   421→          headers: { 'Content-Type': 'application/json' },
   422→        });
   423→      }
   424→      await db.learningMessage.create({
   425→        data: { sessionId, role: 'assistant', content: aiContent, type: 'dialogue' },
   426→      });
   427→      const encoder = new TextEncoder();
   428→      const fallback = new ReadableStream({
   429→        start(controller) {
   430→          controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: aiContent })}\n\n`));
   431→          controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   432→          controller.close();
   433→        },
   434→      });
   435→      return new Response(fallback, {
   436→        headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', Connection: 'keep-alive' },
   437→      });
   438→    }
   439→
   440→    const encoder = new TextEncoder();
   441→    const bufferRef = { buf: '' };
   442→    let fullContent = '';
   443→    let fullThinking = '';
   444→    // Phase tracking — the model emits reasoning_content BEFORE content, so we
   445→    // can send an explicit `phase: 'thinking'` event the moment the first
   446→    // reasoning token arrives, and `phase: 'answering'` when the first content
   447→    // token arrives. This lets the client render a thinking animation that
   448→    // lasts the entire reasoning phase, not just the pre-first-token gap.
   449→    let phaseSent: 'thinking' | 'answering' | null = null;
   450→
   451→    // Build a streaming response that:
   452→    //   1. (deep/structured only) runs multi-step reasoning, emitting { step }
   453→    //      events for each intermediate step
   454→    //   2. reads from upstream SDK stream for the final answer
   455→    //   3. parses OpenAI-style SSE deltas (content + reasoning_content)
   456→    //   4. re-emits in our own SSE format:
   457→    //        - { step: { index, total, label, result } } for each reasoning step
   458→    //        - { phase: 'thinking' } once when reasoning starts
   459→    //        - { thinking: string, full: true } on each reasoning chunk
   460→    //        - { phase: 'answering' } once when content starts
   461→    //        - { content: string, full: true } on each content chunk
   462→    //   5. accumulates full content + persists to DB when the stream ends
   463→    const stream = new ReadableStream<Uint8Array>({
   464→      async start(controller) {
   465→        // ── Phase 2/3: Multi-step reasoning for deep/structured modes ──
   466→        //
   467→        // Before streaming the final answer, run intermediate reasoning
   468→        // steps (analyze → reason → ... ) and emit each as SSE events.
   469→        //
   470→        // Phase 3: steps now stream their tokens live via { stepStart } +
   471→        // { stepToken } events, and citations are sent via { citations }.
   472→        let multiStepContext: string | null = null;
   473→        if (thinkingMode === 'deep' || thinkingMode === 'structured') {
   474→          try {
   475→            const result = await runMultiStepReasoning(
   476→              {
   477→                mode: thinkingMode,
   478→                sessionId,
   479→                message,
   480→                messageHistory: historyMessages || [],
   481→                knowledgeNodes: knowledgeNodes || [],
   482→                teachingMode,
   483→                selectedModel,
   484→              },
   485→              // onStep — called when a step completes with its full result
   486→              (step) => {
   487→                controller.enqueue(
   488→                  encoder.encode(`data: ${JSON.stringify({ step })}\n\n`)
   489→                );
   490→              },
   491→              // onStepStart — called when a step begins (frontend shows indicator)
   492→              (index, total, label) => {
   493→                controller.enqueue(
   494→                  encoder.encode(`data: ${JSON.stringify({ stepStart: { index, total, label } })}\n\n`)
   495→                );
   496→              },
   497→              // onStepToken — called for each token as it streams in
   498→              (index, token) => {
   499→                controller.enqueue(
   500→                  encoder.encode(`data: ${JSON.stringify({ stepToken: { index, token } })}\n\n`)
   501→                );
   502→              },
   503→            );
   504→            multiStepContext = result?.finalThinking ?? null;
   505→
   506→            // Phase 3: send citations so the frontend can render [1] [2] refs
   507→            if (result?.citations && result.citations.length > 0) {
   508→              controller.enqueue(
   509→                encoder.encode(`data: ${JSON.stringify({ citations: result.citations })}\n\n`)
   510→              );
   511→            }
   512→
   513→            // Phase 3: send reasoning metrics
   514→            if (result?.metrics) {
   515→              controller.enqueue(
   516→                encoder.encode(`data: ${JSON.stringify({ metrics: result.metrics })}\n\n`)
   517→              );
   518→            }
   519→          } catch (err) {
   520→            console.error('Multi-step reasoning failed, falling back to single-step:', err);
   521→          }
   522→        }
   523→
   524→        const reader = upstream.getReader();
   525→        try {
   526→          while (true) {
   527→            const { done, value } = await reader.read();
   528→            if (done) break;
   529→            const chunkStr = new TextDecoder().decode(value, { stream: true });
   530→            const pieces = parseUpstreamSse(chunkStr, bufferRef);
   531→            for (const piece of pieces) {
   532→              if (piece.kind === 'thinking') {
   533→                fullThinking += piece.text;
   534→                // Emit phase signal once, on the first reasoning token.
   535→                if (phaseSent !== 'thinking') {
   536→                  phaseSent = 'thinking';
   537→                  controller.enqueue(
   538→                    encoder.encode(`data: ${JSON.stringify({ phase: 'thinking' })}\n\n`)
   539→                  );
   540→                }
   541→                // Forward the accumulated thinking text (sanitized) so the
   542→                // client COULD display a collapsible "reasoning" panel later.
   543→                controller.enqueue(
   544→                  encoder.encode(
   545→                    `data: ${JSON.stringify({ thinking: stripEmoji(fullThinking), full: true })}\n\n`
   546→                  )
   547→                );
   548→              } else {
   549→                fullContent += piece.text;
   550→                if (phaseSent !== 'answering') {
   551→                  phaseSent = 'answering';
   552→                  controller.enqueue(
   553→                    encoder.encode(`data: ${JSON.stringify({ phase: 'answering' })}\n\n`)
   554→                  );
   555→                }
   556→                // Re-emit the *sanitized* accumulated content as a single delta.
   557→                // We strip emoji server-side too so even raw SSE consumers see a
   558→                // monochrome stream; the client also strips as a belt-and-braces.
   559→                const sanitizedAccum = stripEmoji(fullContent);
   560→                controller.enqueue(
   561→                  encoder.encode(`data: ${JSON.stringify({ content: sanitizedAccum, full: true })}\n\n`)
   562→                );
   563→              }
   564→            }
   565→          }
   566→          // Flush any trailing buffered event
   567→          if (bufferRef.buf.trim()) {
   568→            const tail = parseUpstreamSse('\n\n', bufferRef);
   569→            for (const piece of tail) {
   570→              if (piece.kind === 'thinking') {
   571→                fullThinking += piece.text;
   572→              } else {
   573→                fullContent += piece.text;
   574→              }
   575→            }
   576→            if (fullContent) {
   577→              const sanitizedAccum = stripEmoji(fullContent);
   578→              controller.enqueue(
   579→                encoder.encode(`data: ${JSON.stringify({ content: sanitizedAccum, full: true })}\n\n`)
   580→              );
   581→            }
   582→          }
   583→          controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   584→        } catch (err) {
   585→          console.error('Stream read error:', err);
   586→          // Try to send an error event then close gracefully
   587→          controller.enqueue(
   588→            encoder.encode(
   589→              `data: ${JSON.stringify({ error: 'stream_interrupted' })}\n\n`
   590→            )
   591→          );
   592→        } finally {
   593→          controller.close();
   594→          // Persist the accumulated AI message (only if we got something).
   595→          // Sanitize once more before persisting so the stored historical
   596→          // content is guaranteed emoji-free.
   597→          const persistedContent = stripEmoji(fullContent);
   598→          if (persistedContent.trim().length > 0) {
   599→            try {
   600→              await db.learningMessage.create({
   601→                data: {
   602→                  sessionId,
   603→                  role: 'assistant',
   604→                  content: persistedContent,
   605→                  type: 'dialogue',
   606→                  // Persist the reasoning trace alongside the answer so future
   607→                  // turns can reference it (and the UI can show a "查看推理"
   608→                  // collapsible on historical messages). For deep/structured
   609→                  // modes, prepend the multi-step reasoning context.
   610→                  thinking: [
   611→                    multiStepContext ? `## 多步推理上下文\n${multiStepContext}` : '',
   612→                    fullThinking ? `## 模型推理\n${stripEmoji(fullThinking)}` : '',
   613→                  ].filter(Boolean).join('\n\n').slice(0, 8000) || null,
   614→                },
   615→              });
   616→            } catch (dbErr) {
   617→              console.error('Failed to persist AI message:', dbErr);
   618→            }
   619→          }
   620→        }
   621→      },
   622→      cancel() {
   623→        // Client navigated away; nothing more to do. The DB write in finally
   624→        // still fires for whatever content was accumulated.
   625→      },
   626→    });
   627→
   628→    return new Response(stream, {
   629→      headers: {
   630→        'Content-Type': 'text/event-stream',
   631→        'Cache-Control': 'no-cache',
   632→        Connection: 'keep-alive',
   633→        // Disable proxy buffering so chunks flush to the client immediately
   634→        'X-Accel-Buffering': 'no',
   635→      },
   636→    });
   637→  } catch (error) {
   638→    console.error('Chat API error:', error);
   639→    return new Response(JSON.stringify({ error: 'Failed to generate response' }), {
   640→      status: 500,
   641→      headers: { 'Content-Type': 'application/json' },
   642→    });
   643→  }
