   247→// ─── Achievement Unlock Tracking ────────────────────────────────────────────
   248→// Persisted set of achievement IDs the user has already unlocked, so we can
   249→// detect newly-unlocked achievements across page reloads and trigger toasts.
   250→const UNLOCKED_ACH_KEY = 'mindguide:unlocked-achievements';
   251→function loadUnlockedAchievements(): Set<string> {
   252→  if (typeof window === 'undefined') return new Set();
   253→  try {
   254→    const raw = window.localStorage.getItem(UNLOCKED_ACH_KEY);
   255→    return new Set(raw ? JSON.parse(raw) : []);
   256→  } catch {
   257→    return new Set();
   258→  }
   259→}
   260→function saveUnlockedAchievements(ids: Set<string>) {
   261→  if (typeof window === 'undefined') return;
   262→  try {
   263→    window.localStorage.setItem(UNLOCKED_ACH_KEY, JSON.stringify([...ids]));
   264→  } catch {
   265→    // ignore quota errors
   266→  }
   267→}
   268→// In-memory snapshot so we don't re-read localStorage on every fetchStats
   269→let knownUnlocked = loadUnlockedAchievements();

   271→// ─── Store ──────────────────────────────────────────────────────────────────
   272→
   273→export const useLearningStore = create<LearningStore>((set, get) => ({
   274→  ...initialState,
   275→
   276→  // ── Sessions ──────────────────────────────────────────────────────────────
   277→
   278→  fetchSessions: async () => {
   279→    set({ isLoading: true });
   280→    try {
   281→      const res = await fetch('/api/sessions');
   282→      if (!res.ok) { console.error('fetchSessions failed:', res.status); return; }
   283→      const data = await res.json();
   284→      set({ sessions: data.sessions || [] });
   285→    } catch (error) {
   286→      console.error('Failed to fetch sessions:', error);
   287→    } finally {
   288→      set({ isLoading: false });
   289→    }
   290→  },
   291→
   292→  createSession: async (title: string, topic?: string) => {
   293→    try {
   294→      const res = await fetch('/api/sessions', {
   295→        method: 'POST',
   296→        headers: { 'Content-Type': 'application/json' },
   297→        body: JSON.stringify({ title, topic }),
   298→      });
   299→      if (!res.ok) { console.error('createSession failed:', res.status); return null; }
   300→      const data = await res.json();
   301→      if (data.session) {
   302→        set((state) => ({
   303→          sessions: [data.session, ...state.sessions],
   304→        }));
   305→        await get().selectSession(data.session.id);
   306→        return data.session;
   307→      }
   308→      return null;
   309→    } catch (error) {
   310→      console.error('Failed to create session:', error);
   311→      return null;
   312→    }
   313→  },
   314→
   315→  selectSession: async (id: string) => {
   316→    // Preserve notes save state from previous session if needed
   317→    set({
   318→      currentSessionId: id,
   319→      // Returning to a conversation should always show the chat view, not
   320→      // whatever feature view (notes/stats/etc.) was open before.
   321→      activeFeatureView: null,
   322→      messages: [],
   323→      knowledgeNodes: [],
   324→      references: [],
   325→      courseModules: [],
   326→      isCourseGenerated: false,
   327→      notesContent: '',
   328→      notesSaveStatus: 'idle',
   329→      tasks: [],
   330→      cards: [],
   331→    });
   332→    await Promise.all([
   333→      get().fetchMessages(id),
   334→      get().fetchKnowledgeNodes(id),
   335→      get().fetchReferences(id),
   336→      get().fetchCourse(id),
   337→      get().fetchNotes(id),
   338→      get().fetchTasks(id),
   339→      get().fetchCards(id),
   340→    ]);
   341→  },
   342→
   343→  deleteSession: async (id: string) => {
   344→    try {
   345→      const res = await fetch(`/api/sessions/${id}`, { method: 'DELETE' });
   346→      if (!res.ok) { console.error('deleteSession failed:', res.status); return; }
   347→      const { currentSessionId, sessions } = get();
   348→      const isCurrent = currentSessionId === id;
   349→      set({
   350→        sessions: sessions.filter((s) => s.id !== id),
   351→        currentSessionId: isCurrent ? null : currentSessionId,
   352→        // Also clear related data when deleting current session
   353→        ...(isCurrent ? {
   354→          messages: [],
   355→          knowledgeNodes: [],
   356→          references: [],
   357→          courseModules: [],
   358→          isCourseGenerated: false,
   359→          notesContent: '',
   360→        } : {}),
   361→      });
   362→    } catch (error) {
   363→      console.error('Failed to delete session:', error);
   364→    }
   365→  },
   366→
   367→  updateSessionTitle: async (id: string, title: string) => {
   368→    if (!title.trim()) return;
   369→    try {
   370→      const res = await fetch(`/api/sessions/${id}`, {
   371→        method: 'PATCH',
   372→        headers: { 'Content-Type': 'application/json' },
   373→        body: JSON.stringify({ title: title.trim().slice(0, 100) }),
   374→      });
   375→      if (!res.ok) { console.error('updateSessionTitle failed:', res.status); return; }
   376→      set((state) => ({
   377→        sessions: state.sessions.map((s) =>
   378→          s.id === id ? { ...s, title: title.trim().slice(0, 100) } : s
   379→        ),
   380→      }));
   381→    } catch (error) {
   382→      console.error('Failed to update session title:', error);
   383→    }
   384→  },
   385→
   386→  // ── Messages ───────────────────────────────────────────────────────────────
   387→
   388→  sendMessage: async (content: string) => {
   389→    const { currentSessionId } = get();
   390→    if (!currentSessionId) return;
   391→
   392→    // Optimistic: show user message immediately
   393→    const userMessage: LearningMessage = {
   394→      id: `temp-${Date.now()}`,
   395→      sessionId: currentSessionId,
   396→      role: 'user',
   397→      content,
   398→      type: 'dialogue',
   399→      createdAt: new Date().toISOString(),
   400→    };
   401→
   402→    set((state) => ({
   403→      messages: [...state.messages, userMessage],
   404→      isStreaming: true,
   405→      streamingContent: '',
   406→    }));
   407→
   408→    try {
   409→      const res = await fetch('/api/chat', {
   410→        method: 'POST',
   411→        headers: { 'Content-Type': 'application/json' },
   412→        body: JSON.stringify({
   413→          sessionId: currentSessionId,
   414→          message: content,
   415→          messages: get().messages.map((m) => ({
   416→            role: m.role,
   417→            content: m.content,
   418→            type: m.type,
   419→          })),
   420→          knowledgeNodes: get().knowledgeNodes.map((n) => ({
   421→            title: n.title,
   422→            content: n.content,
   423→            category: n.category,
   424→            mastered: n.mastered,
   425→          })),
   426→        }),
   427→      });
   428→
   429→      if (!res.ok) {
   430→        throw new Error(`Chat API returned ${res.status}`);
   431→      }
   432→
   433→      if (!res.body) {
   434→        throw new Error('No response body');
   435→      }
   436→
   437→      const reader = res.body.getReader();
   438→      const decoder = new TextDecoder();
   439→      let fullContent = '';
   440→
   441→      while (true) {
   442→        const { done, value } = await reader.read();
   443→        if (done) break;
   444→
   445→        const chunk = decoder.decode(value, { stream: true });
   446→        const lines = chunk.split('\n');
   447→
   448→        for (const line of lines) {
   449→          if (line.startsWith('data: ')) {
   450→            const data = line.slice(6);
   451→            if (data === '[DONE]') continue;
   452→            try {
   453→              const parsed = JSON.parse(data);
   454→              if (parsed.content) {
   455→                if (parsed.full) {
   456→                  // Server sends the full sanitized accumulator; replace
   457→                  // rather than append to avoid drift.
   458→                  fullContent = parsed.content;
   459→                } else {
   460→                  fullContent += parsed.content;
   461→                }
   462→                // Strip emoji / decorative symbols so the live-streaming
   463→                // bubble stays thesis-monochrome even if the model emits
   464→                // them despite the system prompt.
   465→                set({ streamingContent: stripEmoji(fullContent) });
   466→              }
   467→            } catch {
   468→              // Skip malformed JSON chunks
   469→            }
   470→          }
   471→        }
   472→      }
   473→
   474→      // Re-fetch from DB to get server-side IDs (prevents duplicates)
   475→      set({
   476→        isStreaming: false,
   477→        streamingContent: '',
   478→      });
   479→      await get().fetchMessages(currentSessionId);
   480→
   481→      // Refresh knowledge nodes and references
   482→      get().fetchKnowledgeNodes(currentSessionId);
   483→      get().fetchReferences(currentSessionId);
   484→
   485→      // Refresh global stats in the background (achievement progress may have changed)
   486→      get().fetchStats();
   487→
   488→    } catch (error) {
   489→      console.error('Failed to send message:', error);
   490→      set({ isStreaming: false, streamingContent: '' });
   491→    }
   492→  },
   493→
   494→  fetchMessages: async (sessionId: string) => {
   495→    set({ isLoadingMessages: true });
   496→    try {
   497→      const res = await fetch(`/api/sessions/${sessionId}/messages`);
   498→      if (!res.ok) { console.error('fetchMessages failed:', res.status); return; }
   499→      const data = await res.json();
   500→      // Sanitize historical assistant messages too, so old emoji-laden
   501→      // replies (generated before the system-prompt rule) also appear
   502→      // monochrome. User messages are left untouched.
   503→      const cleaned = (data.messages || []).map((m: { role: string; content: string }) =>
   504→        m.role === 'assistant' ? { ...m, content: stripEmoji(m.content || '') } : m
   505→      );
   506→      set({ messages: cleaned });
   507→    } catch (error) {
   508→      console.error('Failed to fetch messages:', error);
   509→    } finally {
   510→      set({ isLoadingMessages: false });
   511→    }
   512→  },
   513→
   514→  // ── Knowledge ────────────────────────────────────────────────────────────
   515→
   516→  fetchKnowledgeNodes: async (sessionId: string) => {
   517→    try {
   518→      const res = await fetch(`/api/sessions/${sessionId}/knowledge`);
   519→      if (!res.ok) { console.error('fetchKnowledgeNodes failed:', res.status); return; }
   520→      const data = await res.json();
   521→      set({ knowledgeNodes: data.nodes || [] });
   522→    } catch (error) {
   523→      console.error('Failed to fetch knowledge nodes:', error);
   524→    }
   525→  },
   526→
   527→  fetchReferences: async (sessionId: string) => {
   528→    try {
   529→      const res = await fetch(`/api/sessions/${sessionId}/references`);
   530→      if (!res.ok) { console.error('fetchReferences failed:', res.status); return; }
   531→      const data = await res.json();
   532→      set({ references: data.references || [] });
   533→    } catch (error) {
   534→      console.error('Failed to fetch references:', error);
   535→    }
   536→  },
   537→
   538→  toggleKnowledgeMastered: async (nodeId: string) => {
   539→    // Optimistic update
   540→    const prevNodes = get().knowledgeNodes;
   541→    set((state) => ({
   542→      knowledgeNodes: state.knowledgeNodes.map((n) =>
   543→        n.id === nodeId ? { ...n, mastered: !n.mastered } : n
   544→      ),
   545→    }));
   546→    try {
   547→      const res = await fetch(`/api/knowledge/${nodeId}`, {
   548→        method: 'PATCH',
   549→        headers: { 'Content-Type': 'application/json' },
   550→      });
   551→      if (!res.ok) {
   552→        // Rollback on failure
   553→        set({ knowledgeNodes: prevNodes });
   554→        console.error('toggleKnowledgeMastered failed:', res.status);
   555→      } else {
   556→        // Refresh global stats (mastery counts may have changed)
   557→        get().fetchStats();
   558→      }
   559→    } catch (error) {
   560→      // Rollback on network error
   561→      set({ knowledgeNodes: prevNodes });
   562→      console.error('Failed to update knowledge node:', error);
   563→    }
   564→  },
   565→
   566→  // ── Course ────────────────────────────────────────────────────────────────
   567→
   568→  setCoursePanelOpen: (open: boolean) => set({ coursePanelOpen: open }),
   569→  setActiveLessonId: (id: string | null) => set({ activeLessonId: id }),
   570→
   571→  fetchCourse: async (sessionId: string) => {
   572→    try {
   573→      const res = await fetch(`/api/sessions/${sessionId}/course`);
   574→      if (!res.ok) { console.error('fetchCourse failed:', res.status); return; }
   575→      const data = await res.json();
   576→      const modules: CourseModule[] = (data.modules || []).map((m: any) => ({
   577→        id: m.id,
   578→        sessionId: m.sessionId,
   579→        title: m.title,
   580→        order: m.order,
   581→        lessons: (m.lessons || []).map((l: any) => ({
   582→          id: l.id,
   583→          moduleId: l.moduleId,
   584→          title: l.title,
   585→          type: l.type,
   586→          duration: l.duration,
   587→          status: l.status,
   588→          content: l.content,
   589→          order: l.order,
   590→        })),
   591→      }));
   592→      set({
   593→        courseModules: modules,
   594→        isCourseGenerated: modules.length > 0,
   595→      });
   596→    } catch (error) {
   597→      console.error('Failed to fetch course:', error);
   598→    }
   599→  },
   600→
   601→  generateCourse: async () => {
   602→    const state = get();
   603→    if (!state.currentSessionId || state.isGeneratingCourse) return;
   604→    set({ isGeneratingCourse: true });
   605→    try {
   606→      const res = await fetch('/api/course/generate', {
   607→        method: 'POST',
   608→        headers: { 'Content-Type': 'application/json' },
   609→        body: JSON.stringify({
   610→          sessionId: state.currentSessionId,
   611→          messages: state.messages.map(m => ({ role: m.role, content: m.content, type: m.type })),
   612→          knowledgeNodes: state.knowledgeNodes.map(n => ({
   613→            title: n.title, content: n.content, category: n.category,
   614→            importance: n.importance, mastered: n.mastered,
   615→          })),
   616→        }),
   617→      });

   647→  updateLessonStatus: (moduleId: string, lessonId: string, status: CourseLesson['status']) => {
   648→    const { currentSessionId, courseModules } = get();
   649→    if (!currentSessionId) return;
   650→
   651→    // Find the lesson to persist its new status
   652→    const targetModule = courseModules.find(m => m.id === moduleId);

   654→    if (!targetLesson) return;
   655→
   656→    // Optimistic local update (auto-unlock next lesson if completed)
   657→    set((state) => ({
   658→      courseModules: state.courseModules.map(m => {
   659→        if (m.id !== moduleId) return m;
   660→        const updatedLessons = m.lessons.map(l => {
   661→          if (l.id !== lessonId) return l;
   662→          return { ...l, status };
   663→        });
   664→        if (status === 'completed') {
   665→          const completedIdx = updatedLessons.findIndex(l => l.id === lessonId);
   666→          if (completedIdx >= 0 && completedIdx + 1 < updatedLessons.length) {
   667→            const nextLesson = updatedLessons[completedIdx + 1];
   668→            if (nextLesson.status === 'locked') {
   669→              updatedLessons[completedIdx + 1] = { ...nextLesson, status: 'available' };
   670→              // Persist the auto-unlock too
   671→              fetch(`/api/sessions/${currentSessionId}/course`, {
   672→                method: 'PATCH',
   673→                headers: { 'Content-Type': 'application/json' },
   674→                body: JSON.stringify({ lessonId: nextLesson.id, status: 'available' }),
   675→              }).catch(err => console.error('Auto-unlock persist failed:', err));
   676→            }
   677→          }
   678→        }
   679→        return { ...m, lessons: updatedLessons };
   680→      }),
   681→    }));
   682→
   683→    // Persist the user-triggered status change
   684→    fetch(`/api/sessions/${currentSessionId}/course`, {
   685→      method: 'PATCH',
   686→      headers: { 'Content-Type': 'application/json' },
   687→      body: JSON.stringify({ lessonId, status }),
   688→    }).catch(err => console.error('Lesson status persist failed:', err));
   689→
   690→    // Stats may need refresh (for "depth thinker" achievement)
   691→    get().fetchStats();

   756→  fetchStats: async () => {
   757→    set({ isLoadingStats: true });
   758→    try {
   759→      const res = await fetch('/api/stats');
   760→      if (!res.ok) { console.error('fetchStats failed:', res.status); return; }
   761→      const data = await res.json();
   762→      const newAchievements = data.achievements || [];
   763→      set({
   764→        stats: data.totals,
   765→        weeklyActivity: data.weeklyActivity || [],
   766→        achievements: newAchievements,
   767→      });
   768→
   769→      // Detect newly-unlocked achievements and fire celebratory toasts.
   770→      // We only consider it a "new unlock" if the achievement transitioned
   771→      // from locked → unlocked since the last fetchStats.
   772→      const newlyUnlocked = newAchievements.filter(
   773→        (a: Achievement) => a.unlocked && !knownUnlocked.has(a.id)
   774→      );
   775→      if (newlyUnlocked.length > 0) {
   776→        // Update the known set BEFORE firing toasts so a re-render can't double-fire
   777→        const next = new Set(knownUnlocked);
   778→        for (const a of newlyUnlocked) next.add(a.id);
   779→        knownUnlocked = next;
   780→        saveUnlockedAchievements(next);
   781→
   782→        // Fire celebratory toasts. `toast()` is a plain dispatcher from the
   783→        // client-only use-toast hook; the <Toaster /> component mounted in the
   784→        // root layout subscribes and renders them. Safe to call from the store
   785→        // because it only mutates in-memory state + notifies mounted listeners.
   786→        try {
   787→          const { toast } = await import('@/hooks/use-toast');
   788→          for (const a of newlyUnlocked) {
   789→            toast({
   790→              title: `成就解锁：${a.title}`,
   791→              description: a.description,
   792→              duration: 6000,
   793→              className:
   794→                'border-neutral-200 bg-white text-neutral-900 dark:border-neutral-700 dark:bg-neutral-900 dark:text-neutral-100',
   795→            });
   796→          }
   797→        } catch (e) {
   798→          console.warn('Achievement toast skipped (hook unavailable):', e);
   799→        }
   800→      } else {
   801→        // Make sure already-unlocked achievements are recorded (first load)
   802→        const next = new Set(knownUnlocked);
   803→        let changed = false;
   804→        for (const a of newAchievements) {
   805→          if (a.unlocked && !next.has(a.id)) { next.add(a.id); changed = true; }
   806→        }
   807→        if (changed) { knownUnlocked = next; saveUnlockedAchievements(next); }
   808→      }
   809→    } catch (error) {
   810→      console.error('Failed to fetch stats:', error);
   811→    } finally {
   812→      set({ isLoadingStats: false });
   813→    }
   814→  },
