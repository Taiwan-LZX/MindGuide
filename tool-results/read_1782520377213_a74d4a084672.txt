   154→   154→function buildMessageContext(
   155→   155→  userMessages: Array<{ role: string; content: string; type?: string }>,
   156→   156→  knowledgeNodes: Array<{ title: string; content: string; category?: string; mastered: boolean }>,
   157→   157→  kbContext: string,
   158→   158→  teachingMode: string = 'guide',
   159→   159→  thinkingMode: string = 'standard'
   160→   160→) {
   161→   161→  const contextParts: Array<{ role: 'system' | 'user' | 'assistant'; content: string }> = [];
   162→   162→
   163→   163→  const thinkOverlay = THINK_OVERLAYS[thinkingMode] || '';
   164→   164→  contextParts.push({
   165→   165→    role: 'system',
   166→   166→    content: TEACHING_SYSTEM_PROMPT + (MODE_OVERLAYS[teachingMode] || MODE_OVERLAYS.guide) + thinkOverlay,
   167→   167→  });
   168→   168→
   169→   169→  if (knowledgeNodes.length > 0) {
   170→   170→    const masteredNodes = knowledgeNodes.filter((n) => n.mastered);
   171→   171→    const unmasteredNodes = knowledgeNodes.filter((n) => !n.mastered);
   172→   172→
   173→   173→    let knowledgeContext = '## 学习者的知识状态\n\n';
   174→   174→
   175→   175→    if (masteredNodes.length > 0) {
   176→   176→      knowledgeContext += '已掌握的知识点：\n';
   177→   177→      masteredNodes.forEach((n) => {
   178→   178→        knowledgeContext += `- ${n.title}: ${n.content.slice(0, 100)}...\n`;
   179→   179→      });
   180→   180→    }
   181→   181→
   182→   182→    if (unmasteredNodes.length > 0) {
   183→   183→      knowledgeContext += '\n正在学习/未掌握的知识点：\n';
   184→   184→      unmasteredNodes.forEach((n) => {
   185→   185→        knowledgeContext += `- ${n.title} (${n.category || '概念'}): ${n.content.slice(0, 100)}...\n`;
   186→   186→      });
   187→   187→    }
   188→   188→
   189→   189→    contextParts.push({
   190→   190→      role: 'system',
   191→   191→      content: knowledgeContext,
   192→   192→    });
   193→   193→  }
   194→   194→
   195→   195→  // ─── Knowledge base: RAG-retrieved passages ──────────────────────────────
   196→   196→  //
   197→   197→  // Instead of brute-force injecting the first 20k chars of all materials,
   198→   198→  // we retrieve the top-K passages most relevant to the learner's latest
   199→   199→  // message. This gives the model focused, cited context (Lost in the Middle
   200→   200→  // mitigation — Liu et al. 2023).
   201→   201→  if (kbContext) {
   202→   202→    contextParts.push({
   203→   203→      role: 'system',
   204→   204→      content: kbContext,
   205→   205→    });
   206→   206→  }
   207→   207→
   208→   208→  // ─── Token-budget-aware history compression ──────────────────────────────
   209→   209→  //
   210→   210→  // Older versions just took the last 20 messages verbatim. That overflows
   211→   211→  // the context window on long conversations (each turn can be 1-4k tokens),
   212→   212→  // causing the API to truncate silently or reject the request. We now:
   213→   213→  //
   214→   214→  //   1. Estimate tokens per message (~chars/4 for CJK + ASCII mix).
   215→   215→  //   2. Walk from the MOST RECENT message backwards, accumulating until we
   216→   216→  //      hit a soft budget (RESERVED_TOKENS for system+KB+reply headroom).
   217→   217→  //   3. If even the recent messages overflow, we keep the most recent N
   218→   218→  //      fully and FOLD older messages into a one-line summary each
   219→   219→  //      ("用户问了 X，助教回答了 Y 的要点") so the thread stays coherent
   220→   220→  //      without blowing the budget.
   221→   221→  //   4. The very first user turn is ALWAYS kept in full (it sets the topic).
   222→   222→  //
   223→   223→  // This is a pragmatic middle ground between "drop everything old" (loses
   224→   224→  // topic context) and "keep everything" (overflows). RAG already handles
   225→   225→  // long-term knowledge retrieval; this compression only governs the
   226→   226→  // conversational thread.
   227→   227→
   228→   228→  const MODEL_CONTEXT_WINDOW = 200_000;
   229→   229→  const RESERVED_FOR_SYSTEM_KB_REPLY = 24_000; // headroom for system + KB + the new reply
   230→   230→  const historyBudget = MODEL_CONTEXT_WINDOW - RESERVED_FOR_SYSTEM_KB_REPLY;
   231→   231→
   232→   232→  // Rough token estimator: CJK chars ≈ 1 token each, ASCII ≈ 0.25 tokens/char.
   233→   233→  const estTokens = (s: string) => {
   234→   234→    let cjk = 0, other = 0;
   235→   235→    for (const ch of s) {
   236→   236→      const code = ch.codePointAt(0) || 0;
   237→   237→      if ((code >= 0x3000 && code <= 0x9fff) || (code >= 0xff00 && code <= 0xffef)) cjk++;
   238→   238→      else other++;
   239→   239→    }
   240→   240→    return cjk + Math.ceil(other / 4);
   241→   241→  };
   242→   242→
   243→   243→  // Pre-compute per-message token estimates.
   244→   244→  const tagged = userMessages.map((m, i) => ({ msg: m, idx: i, tokens: estTokens(m.content) }));
   245→   245→
   246→   246→  // Always keep the first user turn (topic anchor) — it's almost always short.
   247→   247→  const firstUserTurn = tagged.find(t => t.msg.role === 'user');
   248→   248→  const firstKeepIds = new Set<number>(firstUserTurn ? [firstUserTurn.idx] : []);
   249→   249→
   250→   250→  // Walk backwards from the latest, accumulating full messages until budget.
   251→   251→  const keptFull: typeof tagged = [];
   252→   252→  let usedTokens = firstUserTurn ? firstUserTurn.tokens : 0;
   253→   253→  for (let i = tagged.length - 1; i >= 0; i--) {
   254→   254→    const t = tagged[i];
   255→   255→    if (firstKeepIds.has(i)) continue; // already counted
   256→   256→    if (usedTokens + t.tokens > historyBudget) break;
   257→   257→    keptFull.unshift(t);
   258→   258→    usedTokens += t.tokens;
   259→   259→  }
   260→   260→  // Ensure first turn is in the kept set (at the front).
   261→   261→  if (firstUserTurn) keptFull.unshift(firstUserTurn);
   262→   262→
   263→   263→  // Anything older than keptFull that isn't the first turn gets folded.
   264→   264→  const keptIdx = new Set(keptFull.map(t => t.idx));
   265→   265→  const folded = tagged.filter(t => !keptIdx.has(t.idx));
   266→   266→
   267→   267→  // Build a single system summary of folded turns (if any).
   268→   268→  if (folded.length > 0) {
   269→   269→    const summaryLines = folded.map(t => {
   270→   270→      const role = t.msg.role === 'assistant' ? '助教' : '学习者';
   271→   271→      const snippet = t.msg.content.replace(/\s+/g, ' ').slice(0, 60);
   272→   272→      return `- ${role}：${snippet}`;
   273→   273→    });
   274→   274→    contextParts.push({
   275→   275→      role: 'system',
   276→   276→      content: `## 更早的对话摘要（已折叠以节省上下文）\n${summaryLines.join('\n')}`,
   277→   277→    });
   278→   278→  }
   279→   279→
   280→   280→  // Normalize roles, then MERGE consecutive same-role messages into one.
   281→   281→  // The ZAI Chat Completions API rejects message arrays where two adjacent
   282→   282→  // turns share the same role (error code 1214 "messages 参数非法").
   283→   283→  // This happens naturally when the learner sends two messages in a row, or
   284→   284→  // when an assistant turn was saved in two pieces — so we coalesce them
   285→   285→  // with a blank-line separator before sending.
   286→   286→  for (const t of keptFull) {
   287→   287→    const role = (t.msg.role === 'assistant' ? 'assistant' : t.msg.role === 'system' ? 'system' : 'user') as
   288→   288→      | 'system'
   289→   289→      | 'user'
   290→   290→      | 'assistant';
   291→   291→    const last = contextParts[contextParts.length - 1];
   292→   292→    if (last && last.role === role) {
   293→   293→      last.content = `${last.content}\n\n${t.msg.content}`;
   294→   294→    } else {
   295→   295→      contextParts.push({ role, content: t.msg.content });
   296→   296→    }
   297→   297→  }
   298→   298→
   299→   299→  // Guarantee the conversation ends with a user turn — the API requires the
   300→   300→  // final message to be from the user when requesting a completion.
   301→   301→  while (contextParts.length > 1 && contextParts[contextParts.length - 1].role !== 'user') {
   302→   302→    // Drop trailing assistant/system turns that have no user follow-up
   303→   303→    // (rare, but avoids "messages must end with user" errors).
   304→   304→    if (contextParts[contextParts.length - 1].role === 'system') break;
   305→   305→    contextParts.pop();
   306→   306→  }
   307→   307→
   308→   308→  return contextParts;
   309→   309→}

   311→   311→// Parse an upstream SSE byte chunk and extract delta content pieces.
   312→   312→// Buffers incomplete lines and returns them via the carried-over buffer string.
   313→   313→//
   314→   314→// Returns an array of { kind: 'thinking' | 'content', text } tokens — the GLM
   315→   315→// chat-completions stream emits `delta.reasoning_content` during the model's
   316→   316→// internal reasoning phase (only when thinking is enabled) and
   317→   317→// `delta.content` during the answer phase. Surfacing both lets the client
   318→   318→// render a proper "思考中" animation that persists for the WHOLE reasoning
   319→   319→// phase, not just the pre-first-token gap.
   320→   320→function parseUpstreamSse(
   321→   321→  chunkStr: string,
   322→   322→  bufferRef: { buf: string }
   323→   323→): Array<{ kind: 'thinking' | 'content'; text: string }> {
   324→   324→  bufferRef.buf += chunkStr;
   325→   325→  const pieces: Array<{ kind: 'thinking' | 'content'; text: string }> = [];
   326→   326→  // SSE events are separated by a blank line (\n\n)
   327→   327→  const events = bufferRef.buf.split('\n\n');
   328→   328→  // The last element is the incomplete tail (no trailing \n\n yet)
   329→   329→  bufferRef.buf = events.pop() || '';
   330→   330→
   331→   331→  for (const evt of events) {
   332→   332→    const lines = evt.split('\n');
   333→   333→    for (const line of lines) {
