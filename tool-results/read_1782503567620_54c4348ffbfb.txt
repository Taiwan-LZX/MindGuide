     9→// Moved here from FeatureView because page.tsx's conditional render
    10→// synchronously unmounted FeatureView before AnimatePresence could fire exit.
    11→//
    12→// Direction semantics:
    13→//   dir = +1  → forward (welcome → feature, or feature → different feature)
    14→//   dir = -1  → backward (feature → welcome)
    15→//
    16→// Easing rationale (post user-feedback "关闭的过渡动画无帧数直接闪现"):
    17→//   · Exit previously used ease-in [0.4, 0, 1, 1] which keeps opacity near 1
    18→//     for the first 40% of duration. Combined with ~50ms React commit delay,
    19→//     users saw 195ms of "nothing happening" then a sudden vanish — the
    20→//     "instant close" perception.
    21→//   · Switched exit to ease-out [0.16, 1, 0.3, 1] (snoozeOut): visible motion
    22→//     starts in the very first frame, then gently trails off. The user
    23→//     immediately sees the panel leaving, eliminating the dead-time window.
    24→//   · Entry keeps spring physics — overshoot + settle reads as "arriving".
    25→//   · Per-property transition split: opacity fades slightly faster than
    26→//     transform, so the user perceives "fading out while drifting away"
    27→//     instead of "transform completing then opacity cutting".
    28→

    92→// ─── Animation Variants ─────────────────────────────────────────────────────
    93→//
    94→// Personality: "journey" — interface switching carries directional intent.
    95→// The user either *enters* a feature (forward, from welcome) or *returns* from
    96→// one (backward, to welcome). The motion respects that direction so the
    97→// spatial metaphor holds: forward = pushed from the right (depth entry),
    98→// backward = pushed from the left (depth return).
    99→//
   100→// Differentiation vs. other surfaces:
   101→//  · Quick menu (command): 380/30/0.6 — snappy utility, no depth cue.
   102→//  · More features (discovery): 280/26/0.9 — soft, overshoot ~2%, ~520ms.
   103→//  · Settings (ceremony): 200/24/1.0 — heavy, ~700ms, deliberate.
   104→//  · Feature page (journey): 220/26/0.9 — purposeful, ~620ms, with a clear
   105→//    directional x push + a tiny scale (0.97 → 1) that reads as depth.
   106→//
   107→// `custom` is the direction: +1 = forward (entering a feature), -1 = backward
   108→// (returning to welcome). The exit mirrors the entry direction so the old
   109→// view leaves the opposite way.
   110→//
   111→// NOTE: pageVariants now lives at the top of this file (exported) and is
   112→// consumed by page.tsx's AnimatePresence. The FeatureView component below
   113→// no longer owns an AnimatePresence — page.tsx does. This fixes the
   114→// "feature view back-exit never animates" bug documented in worklog.md
   115→// anim-refine-003: previously, page.tsx's
   116→//   `{activeFeatureView ? <FeatureView /> : <MainContent />}`
   117→// synchronously unmounted FeatureView (and its internal AnimatePresence)
   118→// before framer-motion could fire the exit variant.

   253→const PRIORITY_LABELS: Record<number, string> = {

   261→function PriorityBar({ value, onChange, disabled }: { value: number; onChange: (v: number) => void; disabled?: boolean }) {
   262→  // 5 stacked segments; the bottom `value` are filled (dark), the rest neutral.
   263→  // Click on segment N sets priority to N.
   264→  return (
   265→    <div
   266→      className="flex flex-col-reverse gap-[2px]"
   267→      role="slider"
   268→      aria-label={`优先级 ${value} / 5`}
   269→      aria-valuenow={value}
   270→      aria-valuemin={1}
   271→      aria-valuemax={5}
   272→    >
   273→      {[1, 2, 3, 4, 5].map((n) => {
   274→        const filled = n <= value;
   275→        return (
   276→          <button
   277→            key={n}
   278→            type="button"
   279→            disabled={disabled}
   280→            onClick={(e) => { e.stopPropagation(); onChange(n); }}
   281→            className={`h-[3px] w-3.5 rounded-[1px] transition-colors ${
   282→              filled
   283→                ? 'bg-neutral-800 dark:bg-neutral-200'
   284→                : 'bg-neutral-200 hover:bg-neutral-300 dark:bg-neutral-700 dark:hover:bg-neutral-600'
   285→            } ${disabled ? 'cursor-default opacity-60' : 'cursor-pointer'}`}
   286→            aria-label={`设为优先级 ${n}`}
   287→          />
   288→        );
   289→      })}
   290→    </div>
   291→  );
   292→}

   294→function TaskPlannerView({ scrollRef }: { scrollRef: React.RefObject<HTMLDivElement | null> }) {
   295→  const { tasks, addTask, toggleTask, deleteTask, setTaskPriority, reorderTasks, isLoadingTasks } = useLearningStore();
   296→  const [newTask, setNewTask] = React.useState('');
   297→  const [submitting, setSubmitting] = React.useState(false);
   298→  const [newPriority, setNewPriority] = React.useState(3);
   299→  const [draggingId, setDraggingId] = React.useState<string | null>(null);
   300→  const [dragOverId, setDragOverId] = React.useState<string | null>(null);
   301→
   302→  const submit = React.useCallback(async () => {
   303→    const t = newTask.trim();
   304→    if (!t || submitting) return;
   305→    setSubmitting(true);
   306→    setNewTask('');
   307→    await addTask(t, newPriority);
   308→    setSubmitting(false);
   309→  }, [newTask, submitting, addTask, newPriority]);
   310→
   311→  const doneCount = tasks.filter(t => t.done).length;
   312→  const pct = tasks.length > 0 ? (doneCount / tasks.length) * 100 : 0;
   313→
   314→  // Sort by order (fallback to createdAt)
   315→  const sortedTasks = React.useMemo(() => {
   316→    return [...tasks].sort((a, b) => {
   317→      if (a.order !== b.order) return a.order - b.order;
   318→      return new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime();
   319→    });
   320→  }, [tasks]);
   321→
   322→  const handleDrop = (targetId: string) => {
   323→    if (!draggingId || draggingId === targetId) {
   324→      setDraggingId(null);
   325→      setDragOverId(null);
   326→      return;
   327→    }
   328→    const ids = sortedTasks.map(t => t.id);
   329→    const fromIdx = ids.indexOf(draggingId);
   330→    const toIdx = ids.indexOf(targetId);
   331→    if (fromIdx < 0 || toIdx < 0) {
   332→      setDraggingId(null);
   333→      setDragOverId(null);
   334→      return;
   335→    }
   336→    ids.splice(fromIdx, 1);
   337→    ids.splice(toIdx, 0, draggingId);
   338→    void reorderTasks(ids);
   339→    setDraggingId(null);
   340→    setDragOverId(null);
   341→  };
   342→
   343→  return (
   344→    <>
   345→      <FeatureHeader title="任务规划" icon={Check} color="bg-neutral-100 text-neutral-600 dark:bg-neutral-800 dark:text-neutral-400" />
   346→      <div ref={scrollRef} className="flex-1 overflow-y-auto custom-scrollbar">
   347→        <div className="mx-auto max-w-[600px] px-6 py-5">
   348→          {/* Progress */}
   349→          <motion.div
   350→            initial={{ opacity: 0, y: 8 }}
   351→            animate={{ opacity: 1, y: 0 }}
   352→            className="mb-5 flex items-center justify-between text-[13px]"
   353→          >
   354→            <span className="text-neutral-500 dark:text-neutral-400">进度 {doneCount}/{tasks.length}</span>
   355→            <motion.div
   356→              className="h-1.5 w-32 overflow-hidden rounded-full bg-neutral-100 dark:bg-neutral-800"
   357→              initial={{ opacity: 0 }}
   358→              animate={{ opacity: 1 }}
   359→              transition={{ delay: 0.2 }}
   360→            >
   361→              {/* Progress fill uses transform: scaleX (composited) instead of
   362→                  width: % (which triggers layout). transformOrigin: left so
   363→                  the bar grows from the left edge. scaleX takes a 0-1 value. */}
   364→              <motion.div
   365→                className="h-full rounded-full bg-neutral-900 dark:bg-white"
   366→                style={{ transformOrigin: 'left' }}
   367→                initial={{ scaleX: 0 }}
   368→                animate={{ scaleX: Math.max(0, Math.min(1, pct / 100)) }}
   369→                transition={{ type: 'spring', stiffness: 300, damping: 25 }}
   370→              />
   371→            </motion.div>
   372→          </motion.div>
   373→
   374→          {/* Add task — with priority picker */}
   375→          <motion.div
   376→            initial={{ opacity: 0, y: 8 }}
   377→            animate={{ opacity: 1, y: 0, transition: { delay: 0.05 } }}
   378→            className="mb-4 rounded-lg border border-neutral-200 bg-white p-2 dark:border-neutral-700 dark:bg-neutral-900"
   379→          >
   380→            <div className="flex items-center gap-2">
   381→              <input
   382→                type="text"
   383→                placeholder="添加学习任务..."
   384→                value={newTask}
   385→                onChange={e => setNewTask(e.target.value)}
   386→                onKeyDown={e => { if (e.key === 'Enter') { void submit(); } }}
   387→                className="h-9 flex-1 bg-transparent px-2 text-[13px] transition-colors duration-150 focus:outline-none dark:text-neutral-200"
   388→              />
   389→              <motion.button
   390→                whileHover={{
   391→                  scale: 1.05,
   392→                  transition: { type: 'spring', stiffness: 400, damping: 22 },
   393→                }}
   394→                whileTap={{
   395→                  scale: 0.94,
   396→                  transition: { type: 'spring', stiffness: 600, damping: 25 },
   397→                }}
   398→                onClick={() => { void submit(); }}
   399→                disabled={submitting}
   400→                className="flex h-9 w-9 items-center justify-center rounded-lg bg-neutral-900 text-white transition-opacity disabled:opacity-50 dark:bg-white dark:text-neutral-900"
   401→              >
   402→                <Plus className="h-4 w-4" />
   403→              </motion.button>
   404→            </div>
   405→            {/* Priority selector for new task */}
   406→            <div className="mt-1.5 flex items-center gap-3 border-t border-neutral-100 px-2 pt-2 dark:border-neutral-800">
   407→              <span className="text-[10px] uppercase tracking-wider text-neutral-400 dark:text-neutral-500">优先级</span>
   408→              <div className="flex gap-1">
   409→                {[1, 2, 3, 4, 5].map(n => (
   410→                  <button
   411→                    key={n}
   412→                    type="button"
   413→                    onClick={() => setNewPriority(n)}
   414→                    className={`h-5 w-5 rounded-md border text-[10px] tabular-nums transition-colors ${
   415→                      newPriority === n
   416→                        ? 'border-neutral-800 bg-neutral-800 text-white dark:border-neutral-200 dark:bg-neutral-200 dark:text-neutral-900'
   417→                        : 'border-neutral-200 text-neutral-400 hover:border-neutral-300 dark:border-neutral-700 dark:text-neutral-500'
   418→                    }`}
   419→                  >
   420→                    {n}
   421→                  </button>
   422→                ))}
   423→              </div>
   424→              <span className="text-[11px] text-neutral-500 dark:text-neutral-400">{PRIORITY_LABELS[newPriority]}</span>
   425→            </div>
   426→          </motion.div>
   427→
   428→          {/* Task list */}
   429→          <AnimatePresence mode="popLayout" initial={false}>
   430→            {isLoadingTasks && tasks.length === 0 && (
   431→              <div className="space-y-1.5">
   432→                {[0, 1, 2].map(i => (
   433→                  <div key={i} className="flex items-center gap-3 rounded-lg bg-white px-3 py-2.5 dark:bg-neutral-900">
   434→                    <div className="h-5 w-5 shrink-0 rounded-md bg-neutral-100 dark:bg-neutral-800" />
   435→                    <div className="h-3 flex-1 rounded bg-neutral-100 dark:bg-neutral-800" style={{ width: `${60 - i * 12}%` }} />
   436→                  </div>
   437→                ))}
   438→              </div>
   439→            )}
   440→            {!isLoadingTasks && tasks.length === 0 && <EmptyState icon={Check} title="暂无任务" description="添加学习任务来规划你的学习路径" />}
   441→            {sortedTasks.map((task, i) => {
   442→              const isDragging = draggingId === task.id;
   443→              const isDragOver = dragOverId === task.id && draggingId !== task.id;
   444→              return (
   445→                <motion.div
   446→                  key={task.id}
   447→                  custom={i}
   448→                  variants={itemVariants}
   449→                  initial="hidden"
   450→                  animate="visible"
   451→                  exit={{ opacity: 0, x: -20, scale: 0.95, transition: { duration: 0.15 } }}
   452→                  layout
   453→                  draggable
   454→                  onDragStart={() => setDraggingId(task.id)}
   455→                  onDragEnd={() => { setDraggingId(null); setDragOverId(null); }}
   456→                  onDragOver={(e) => { e.preventDefault(); setDragOverId(task.id); }}
   457→                  onDrop={() => handleDrop(task.id)}
   458→                  className={`group mb-1.5 flex items-center gap-2 rounded-lg border px-2.5 py-2.5 transition-all ${
   459→                    isDragging ? 'opacity-50' : ''
   460→                  } ${
   461→                    isDragOver ? 'border-neutral-400 bg-neutral-50 dark:border-neutral-500 dark:bg-neutral-800' : 'border-transparent'
   462→                  } ${
   463→                    task.done ? 'bg-neutral-50 dark:bg-neutral-800/50' : 'bg-white hover:bg-neutral-50 dark:bg-neutral-900 dark:hover:bg-neutral-800/50'
   464→                  }`}
   465→                >
   466→                  {/* Drag handle */}
   467→                  <span
   468→                    className="flex h-5 w-3 cursor-grab items-center justify-center text-neutral-300 opacity-0 transition-opacity hover:text-neutral-500 group-hover:opacity-100 active:cursor-grabbing dark:text-neutral-600 dark:hover:text-neutral-300"
   469→                    aria-hidden="true"
   470→                  >
   471→                    <GripVertical className="h-3.5 w-3.5" />

   473→                  {/* Checkbox */}
   474→                  <motion.button
   475→                    whileTap={{
   476→                      scale: 0.8,
   477→                      transition: { type: 'spring', stiffness: 600, damping: 18 },
   478→                    }}
   479→                    onClick={() => { void toggleTask(task.id); }}
   480→                    className={`flex h-5 w-5 shrink-0 items-center justify-center rounded-md border transition-colors ${
   481→                      task.done
   482→                        ? 'border-neutral-700 bg-neutral-700 text-white dark:border-neutral-300 dark:bg-neutral-300 dark:text-neutral-900'
   483→                        : 'border-neutral-300 dark:border-neutral-600'
   484→                    }`}
   485→                  >
   486→                    {task.done && <Check className="h-3 w-3" strokeWidth={3} />}

   488→                  {/* Priority bar — interactive 5-segment slider. We use a

  1645→function OutlinePanel({ materialId }: { materialId: string }) {
